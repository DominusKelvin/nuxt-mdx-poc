# Vue Live Editor Example
import Vue from 'vue'
import Count from '@/components/Count'
import { LiveEditor } from '@/components'
import Button from '@/components/Button'
Vue.component('Button', Button)


#  `Demo` example

<LiveEditor>
{`
<template>
  <div>
    <h3>Count: {{ count }} </h3>
    <Button style="background: tomato;" @click="decrement">
      Decrease
    </Button>
    <Button @click="increment">
      Increase
    </Button>
  </div>
</template>
<script>
  export default {
    name: 'Count',
    data () {
      return {
        count: 12
      }
    },
    methods: {
      increment () {
        this.count++
      },
      decrement () {
        this.count--
      }
    }
  }
<\/script>
`}
</LiveEditor>
